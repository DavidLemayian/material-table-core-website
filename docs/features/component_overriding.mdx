import MaterialTable, { MTableToolbar } from '@material-table/core';

@material-table/core has a flexible API to meet your needs.

### Usage

```jsx
<MaterialTable
    // other props
    components={{
        Toolbar: props => (
            <div style={{ backgroundColor: '#e8eaf5' }}>
                <MTableToolbar {...props} />
            </div>
        )
    }}
/>
```

### Live Code Demo

```jsx live
function ComponentOverriding() {
  const [bgColor, setBgColor] = useState(RAND_COLOR);

  return (
    <div>
      <Button variant="contained" onClick={() => setBgColor(RAND_COLOR)}>
        Change Background Color
      </Button>

      <MaterialTable 
        data={[
          { id: 2, name: "Joe" }, 
          { id: 1, name: "Mary" }
        ]} 
        columns={[
          { field: "id", title: "Id" }, 
          { field: "name", title: "Name" }
        ]} 
        components={{
          // Override default toolbar
          Toolbar: props =>
            <div style={{ backgroundColor: bgColor }}>
              <MTableToolbar {...props} />
            </div>
        }}
        icons={TABLE_ICONS} />
    </div>
  );
}
```
