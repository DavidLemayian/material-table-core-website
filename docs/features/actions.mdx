import MaterialTable from '@material-table/core';

You can add buttons to rows or toolbar by using actions prop. Actions prop takes array of actions.

### Usage

```jsx
<MaterialTable
  // other props
  actions={[
    {
      icon: 'save',
      tooltip: 'Save User',
      onClick: (event, rowData) => {
        // Do save operation
        console.log({ rowData, event });
        alert(rowData);
      }
    }
  ]}
/>
```

### Live Code Demo

```jsx live
function ActionsDemo() {
  return (
    <MaterialTable
      data={[
        { id: 2, name: "Joe" }, 
        { id: 1, name: "Mary" }
      ]} 
      columns={[
        { field: "id", title: "Id" }, 
        { field: "name", title: "Name" }
      ]} 
      actions={[
        {
          icon: () => <SaveIcon />,
          tooltip: 'Save User',
          onClick: (event, rowData) => {
            alert(`Saved User : ${rowData.id} ${rowData.name}`)
          }
        }
      ]}
      icons={TABLE_ICONS} 
    />
  );
}
```
