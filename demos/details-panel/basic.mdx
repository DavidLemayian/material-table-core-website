## Usage

```jsx
import MaterialTable from "@material-table/core";

<MaterialTable
  // ... other props
  onDetailPanelChange={(row, state) => {...}}
  detailPanel={({ rowData }) => {
    return (
      <div
        style={{
          fontSize: 20,
          textAlign: "center",
          height: 100,
        }}
      >
        This is a detailed panel for {rowData.name}
      </div>
    );
  }}
/>;
```
