## Usage

```jsx {6}
import MaterialTable, { MTableToolbar } from '@material-table/core';

<MaterialTable
    // other props
    components={{
        Toolbar: props => (
            <div style={{ backgroundColor: '#e8eaf5' }}>
                <MTableToolbar {...props} />
            </div>
        )
    }}
/>
```
