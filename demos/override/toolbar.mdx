## Usage

```jsx
import MaterialTable, { MTableToolbar } from "@material-table/core";

<MaterialTable
  // ...
  components={{
    Toolbar: (props) => (
      <div style={{ backgroundColor: "#e8eaf5" }}>
        <MTableToolbar {...props} />
      </div>
    ),
  }}
/>;
```

function OverrideToolbar() {

    <div>

        onClick={() => setBgColor(RAND_COLOR)}
        variant="contained"
        color="primary"

      </Button>

      <MaterialTable
        data={DEMO_DATA}
        columns={DEMO_COLS}
        components={{
          Toolbar: (props) => {

              <div style={{ backgroundColor: bgColor }}>
                <MTableToolbar {...props} />
              </div>
            );
          },
        }}
      />
    </div>
  );
}
